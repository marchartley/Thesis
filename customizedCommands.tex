\zzcommand{\Tilde}[1]{
  \StrLen{#1}[\temp]%
  \ifnum\temp>1
    \ThisStyle{%
      \setbox0=\hbox{$\SavedStyle#1$}%
      \stackengine{-.1\LMpt}{$\SavedStyle#1$}{%
        \stretchto{\scaleto{\SavedStyle\mkern.2mu\AC}{.5150\wd0}}{.6\ht0}%
      }{O}{c}{F}{T}{S}%
    }
  \else 
    \OldTilde{#1}
  \fi
}
